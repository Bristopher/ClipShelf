# Gkey Mover v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewrite Gkey Mover from Python/Tkinter to Tauri v2 with a Rust backend and React + shadcn/ui frontend.

**Architecture:** Rust backend handles all state, file watching, hotkeys, sounds, OBS WebSocket, and timer logic via tokio channels. React frontend is a pure view layer that listens to Tauri events and invokes Tauri commands. Communication is strictly event-driven — no polling.

**Tech Stack:** Tauri 2.x, Rust (tokio, notify, rodio, winapi, tokio-tungstenite), React 19 + TypeScript, Vite, Tailwind CSS v4, shadcn/ui

**Spec:** `Docs/specs/2026-04-15-gkey-mover-v2-design.md`

```rust
impl AppConfig {
    /// Get the config file path (next to the executable)
    pub fn config_path() -> PathBuf {
        std::env::current_exe()
            .map(|p| p.parent().unwrap_or(Path::new(".")).to_path_buf())
            .unwrap_or_else(|_| PathBuf::from("."))
            .join("config.toml")
    }

    /// Load config from TOML file, falling back to defaults
    pub fn load() -> Result<Self, Box<dyn std::error::Error>> {
        Self::load_from(&Self::config_path())
    }

    /// Load from a specific path (used by tests)
    pub fn load_from(path: &Path) -> Result<Self, Box<dyn std::error::Error>> {
        if path.exists() {
            let content = fs::read_to_string(path)?;
            let config: AppConfig = toml::from_str(&content)?;
            Ok(config)
        } else {
            Ok(AppConfig::default())
        }
    }

    /// Save config to TOML file
    pub fn save(&self) -> Result<(), Box<dyn std::error::Error>> {
        self.save_to(&Self::config_path())
    }

    /// Save to a specific path (used by tests)
    pub fn save_to(&self, path: &Path) -> Result<(), Box<dyn std::error::Error>> {
        let content = toml::to_string_pretty(self)?;
        fs::write(path, content)?;
        Ok(())
    }

    /// Merge a partial JSON object into the config (for update_config command)
    pub fn merge_partial(&mut self, partial: serde_json::Value) {
        if let serde_json::Value::Object(map) = partial {
            // Serialize current config to JSON, merge, deserialize back
            let mut current = serde_json::to_value(&*self).unwrap();
            if let serde_json::Value::Object(ref mut current_map) = current {
                for (key, value) in map {
                    current_map.insert(key, value);
                }
            }
            if let Ok(merged) = serde_json::from_value::<AppConfig>(current) {
                *self = merged;
            }
        }
    }

    /// Get the timer duration in seconds
    pub fn timer_duration_secs(&self) -> u64 {
        self.timer_duration_ms / 1000
    }

    /// Get initial timer display string (MM:SS)
    pub fn timer_display(&self) -> String {
        let total_secs = self.timer_duration_secs();
        let minutes = total_secs / 60;
        let seconds = total_secs % 60;
        format!("{:02}:{:02}", minutes, seconds)
    }

    /// Get the sort folder path for a given G-key
    pub fn sort_folder_path(&self, gkey: u8) -> PathBuf {
        let folder_name = match gkey {
            1 => &self.g1_bind_folder_name,
            2 => &self.g2_bind_folder_name,
            3 => &self.g3_bind_folder_name,
            _ => &self.g1_bind_folder_name,
        };
        PathBuf::from(&self.videos_folder)
            .join("sort")
            .join("AHK sort")
            .join(folder_name)
    }
}
```

```rust
pub struct AppLogger {

impl AppLogger {
    pub fn new(videos_folder: &str, log_enabled: bool) -> Self {
        let log_dir = if videos_folder.is_empty() {

        } else {
            let dir = PathBuf::from(videos_folder).join("logs");
            fs::create_dir_all(&dir).ok();

            history: Vec::new(),
            display_buffer: Vec::new(),

        }

        let entry = LogEntryPayload {
            timestamp: Local::now().format("%I:%M:%S %p").to_string(),
            level,
            message,
            category,
        };
        self.history.push(entry.clone());
        self.display_buffer.push(entry.clone());
        entry
    }

    /// Write a move/rename entry to the daily log file
    pub fn write_to_file(&self, line: &str) {
        if !self.log_enabled {
            return;
        }
        if let Some(ref log_dir) = self.log_dir {
            let today = Local::now().format("%Y-%m-%d").to_string();
            let filename = format!("ObsMoveLog {}.txt", today);
            let path = log_dir.join(filename);

            if let Ok(mut file) = OpenOptions::new().create(true).append(true).open(path) {
                let _ = writeln!(file, "{}", line);
            }
        }
    }

    /// Clear display buffer (wipe)
    pub fn wipe_display(&mut self) {
        self.display_buffer.clear();
    }

    /// Restore display buffer from full history, filtering out noise
    pub fn restore_display(&mut self) -> Vec<LogEntryPayload> {
        let restored: Vec<LogEntryPayload> = self.history.iter().filter(|entry| {
            !matches!(entry.category, LogCategory::HotkeyPressed)
                && !entry.message.to_lowercase().contains("no current_file")
        }).cloned().collect();

        self.display_buffer = restored.clone();
        restored
    }

    /// Get current display buffer
    pub fn display_entries(&self) -> &[LogEntryPayload] {
        &self.display_buffer
    }

    /// Get full history
    pub fn history(&self) -> &[LogEntryPayload] {
        &self.history
    }
}
```

```rust
/// Map a bind folder name to its tag shorthand
pub fn get_tag_shorthand(bind_folder_name: &str) -> &str {
    match bind_folder_name {
        "!! or ! (G1)" => "!!",
        "odd or checked (G2)" => "CHKD",
        "!!! (G3)" => "!!!",
        _ => bind_folder_name,

/// Insert a tag after the timestamp in a filename.
/// Handles OBS format: `Replay YYYY-MM-DD HH-MM-SS.mp4`
/// Handles ShadowPlay format: `Game YYYY.MM.DD - HH.MM.SS.mp4`
/// Falls back to appending before extension if no timestamp found.
pub fn insert_tag_in_filename(filename: &str, tag: &str) -> String {
    // OBS timestamp pattern
    let obs_re = Regex::new(r"(\d{4}-\d{2}-\d{2} \d{2}-\d{2}-\d{2})").unwrap();
    if let Some(m) = obs_re.find(filename) {
        let end = m.end();
        return format!("{} {}{}", &filename[..end], tag, &filename[end..]);
    }

    // ShadowPlay timestamp pattern
    let sp_re = Regex::new(r"(\d{4}\.\d{2}\.\d{2} - \d{2}\.\d{2}\.\d{2})").unwrap();
    if let Some(m) = sp_re.find(filename) {
        let end = m.end();
        return format!("{} {}{}", &filename[..end], tag, &filename[end..]);
    }

    // Fallback: insert before extension
    if let Some(dot_pos) = filename.rfind('.') {
        format!("{} {}{}", &filename[..dot_pos], tag, &filename[dot_pos..])
    } else {
        format!("{} {}", filename, tag)
    }
}

/// Check if a file is larger than min_size_mb
pub fn is_file_size_valid(path: &Path, min_size_mb: f64) -> bool {
    match fs::metadata(path) {
        Ok(meta) => (meta.len() as f64 / (1024.0 * 1024.0)) > min_size_mb,
        Err(_) => false,
    }
}

/// Get file size in MB
pub fn file_size_mb(path: &Path) -> f64 {
    fs::metadata(path)
        .map(|m| m.len() as f64 / (1024.0 * 1024.0))
        .unwrap_or(0.0)
}

/// Move or rename a file based on the current mode.
/// In folder sort mode: moves to videos_folder/sort/AHK sort/{bind_folder}/
/// In rename-only mode: renames in place with tag inserted after timestamp.
pub fn move_or_rename_file(
    file_path: &Path,
    gkey: u8,
    config: &AppConfig,
) -> Result<MoveResult, String> {
    if !file_path.exists() {
        return Err(format!("File does not exist: {}", file_path.display()));
    }

    let filename = file_path
        .file_name()
        .and_then(|n| n.to_str())
        .ok_or("Invalid filename")?;

    let bind_folder_name = match gkey {
        1 => &config.g1_bind_folder_name,
        2 => &config.g2_bind_folder_name,
        3 => &config.g3_bind_folder_name,
        _ => return Err(format!("Invalid gkey: {}", gkey)),
    };

    let tag = get_tag_shorthand(bind_folder_name);
    let new_filename = insert_tag_in_filename(filename, tag);

    let new_path = if config.disable_file_movesorting {
        // Rename only mode: same directory, new name
        file_path.parent().unwrap_or(Path::new(".")).join(&new_filename)
    } else {
        // Folder sort mode: move to sort directory
        let sort_dir = config.sort_folder_path(gkey);
        fs::create_dir_all(&sort_dir).map_err(|e| format!("Failed to create sort dir: {}", e))?;
        sort_dir.join(&new_filename)
    };

    // Retry logic for locked files
    let mut last_err = String::new();
    let delays = [200, 500, 1000];
    for (attempt, delay_ms) in delays.iter().enumerate() {
        match fs::rename(file_path, &new_path) {
            Ok(()) => {
                return Ok(MoveResult {
                    original_path: file_path.to_path_buf(),
                    new_path,
                    tag_applied: tag.to_string(),
                });
            }
            Err(e) => {
                last_err = format!("Attempt {}: {}", attempt + 1, e);
                thread::sleep(Duration::from_millis(*delay_ms));
            }
        }
    }

    Err(format!("Failed to move file after 3 attempts: {}", last_err))
}

/// Rename a file by appending user text after the filename.
pub fn rename_file_with_text(
    file_path: &Path,
    text: &str,
) -> Result<MoveResult, String> {
    if !file_path.exists() {
        return Err(format!("File does not exist: {}", file_path.display()));
    }

    let stem = file_path.file_stem().and_then(|s| s.to_str()).ok_or("Invalid filename")?;
    let ext = file_path.extension().and_then(|e| e.to_str()).unwrap_or("");
    let parent = file_path.parent().unwrap_or(Path::new("."));

    let new_filename = if ext.is_empty() {
        format!("{} - {}", stem, text)
    } else {
        format!("{} - {}.{}", stem, text, ext)
    };

    let new_path = parent.join(&new_filename);
    fs::rename(file_path, &new_path).map_err(|e| format!("Rename failed: {}", e))?;

    Ok(MoveResult {
        original_path: file_path.to_path_buf(),
        new_path,
        tag_applied: text.to_string(),
    })
}
```

```rust
use std::sync::{Arc, Mutex};
use tauri::{AppHandle, Emitter};
use tokio::sync::mpsc;
use tokio::time::{interval, Duration};

use crate::events::TimerTickPayload;

pub enum TimerCommand {
    Start { duration_secs: u32 },
    Stop,
    Reset { duration_secs: u32 },
}

/// Spawn a timer task that listens for commands and emits tick events.
/// Returns a sender for sending commands to the timer.
pub fn spawn_timer(app_handle: AppHandle) -> mpsc::Sender<TimerCommand> {
    let (tx, mut rx) = mpsc::channel::<TimerCommand>(32);

    tokio::spawn(async move {
        let mut total_secs: u32 = 70;
        let mut remaining: u32 = 0;
        let mut running = false;
        let mut tick_interval = interval(Duration::from_secs(1));

        loop {
            tokio::select! {
                cmd = rx.recv() => {
                    match cmd {
                        Some(TimerCommand::Start { duration_secs }) => {
                            total_secs = duration_secs;
                            remaining = duration_secs;
                            running = true;
                        }
                        Some(TimerCommand::Stop) => {
                            running = false;
                        }
                        Some(TimerCommand::Reset { duration_secs }) => {
                            total_secs = duration_secs;
                            remaining = duration_secs;
                            running = false;
                            let _ = app_handle.emit("timer-tick", TimerTickPayload {
                                remaining_secs: remaining,
                                total_secs,
                            });
                        }
                        None => break, // Channel closed
                    }
                }
                _ = tick_interval.tick(), if running => {
                    if remaining > 0 {
                        remaining -= 1;
                        let _ = app_handle.emit("timer-tick", TimerTickPayload {
                            remaining_secs: remaining,
                            total_secs,
                        });
                    }
                    if remaining == 0 && running {
                        running = false;
                        let _ = app_handle.emit("timer-expired", ());
                    }
                }
            }
        }
    });

    tx
}
```

```rust
use notify::{Config, Event, EventKind, RecommendedWatcher, RecursiveMode, Watcher};
use std::path::{Path, PathBuf};
use std::sync::{Arc, Mutex};
use std::time::{Duration, Instant};
use tokio::sync::mpsc;

pub enum WatcherCommand {
    Start { path: PathBuf },
    Stop,
    Restart,
}

pub enum WatcherEvent {
    FileCreated { path: PathBuf },
    StatusChanged { status: String, restart_count: u32, message: Option<String> },
    Error { message: String },
}

/// Spawn the file watcher task.
/// Returns a command sender and an event receiver.
pub fn spawn_watcher() -> (mpsc::Sender<WatcherCommand>, mpsc::Receiver<WatcherEvent>) {
    let (cmd_tx, mut cmd_rx) = mpsc::channel::<WatcherCommand>(32);
    let (event_tx, event_rx) = mpsc::channel::<WatcherEvent>(64);

    tokio::spawn(async move {
        let mut watcher: Option<RecommendedWatcher> = None;
        let mut watch_path: Option<PathBuf> = None;
        let mut restart_count: u32 = 0;
        let last_tick = Arc::new(Mutex::new(Instant::now()));

        // Sleep/resume detection tick
        let sleep_check_tx = event_tx.clone();
        let sleep_last_tick = last_tick.clone();
        let sleep_watch_path: Arc<Mutex<Option<PathBuf>>> = Arc::new(Mutex::new(None));
        let sleep_wp_ref = sleep_watch_path.clone();

        // Spawn sleep detection loop
        let sleep_cmd_tx = cmd_tx.clone();
        tokio::spawn(async move {
            let mut interval = tokio::time::interval(Duration::from_secs(1));
            loop {
                interval.tick().await;
                let mut last = sleep_last_tick.lock().unwrap();
                let delta = last.elapsed();
                *last = Instant::now();

                if delta > Duration::from_secs(10) {
                    let _ = sleep_cmd_tx.send(WatcherCommand::Restart).await;
                }
            }
        });

        while let Some(cmd) = cmd_rx.recv().await {
            match cmd {
                WatcherCommand::Start { path } => {
                    // Stop existing watcher
                    watcher.take();

                    let tx_clone = event_tx.clone();
                    let (notify_tx, mut notify_rx) = mpsc::channel(64);

                    let rt_handle = tokio::runtime::Handle::current();

                    let mut w = RecommendedWatcher::new(
                        move |res: Result<Event, notify::Error>| {
                            match res {
                                Ok(event) => {
                                    if matches!(event.kind, EventKind::Create(_)) {
                                        for path in event.paths {
                                            let ext = path.extension()
                                                .and_then(|e| e.to_str())
                                                .unwrap_or("")
                                                .to_lowercase();
                                            if matches!(ext.as_str(), "mp4" | "mov" | "avi" | "mkv") {
                                                let tx = tx_clone.clone();
                                                let p = path.clone();
                                                rt_handle.spawn(async move {
                                                    // Debounce: wait 200ms for file to finish writing
                                                    tokio::time::sleep(Duration::from_millis(200)).await;
                                                    let _ = tx.send(WatcherEvent::FileCreated { path: p }).await;
                                                });
                                            }
                                        }
                                    }
                                }
                                Err(e) => {
                                    let tx = tx_clone.clone();
                                    let msg = e.to_string();
                                    rt_handle.spawn(async move {
                                        let _ = tx.send(WatcherEvent::Error { message: msg }).await;
                                    });
                                }
                            }
                        },
                        Config::default(),
                    ).map_err(|e| format!("Failed to create watcher: {}", e));

                    match w {
                        Ok(mut w) => {
                            if let Err(e) = w.watch(&path, RecursiveMode::NonRecursive) {
                                let _ = event_tx.send(WatcherEvent::Error {
                                    message: format!("Failed to watch {}: {}", path.display(), e),
                                }).await;
                            } else {
                                let _ = event_tx.send(WatcherEvent::StatusChanged {
                                    status: "running".into(),
                                    restart_count,
                                    message: Some(format!("Watching {}", path.display())),
                                }).await;
                                watch_path = Some(path.clone());
                                *sleep_watch_path.lock().unwrap() = Some(path);
                                watcher = Some(w);
                            }
                        }
                        Err(e) => {
                            let _ = event_tx.send(WatcherEvent::Error { message: e }).await;
                        }
                    }
                }
                WatcherCommand::Stop => {
                    watcher.take();
                    let _ = event_tx.send(WatcherEvent::StatusChanged {
                        status: "stopped".into(),
                        restart_count,
                        message: None,
                    }).await;
                }
                WatcherCommand::Restart => {
                    restart_count += 1;
                    if let Some(path) = watch_path.clone() {
                        watcher.take();

                        let _ = event_tx.send(WatcherEvent::StatusChanged {
                            status: "running".into(),
                            restart_count,
                            message: Some(format!("Restarted (#{}) after sleep/resume", restart_count)),
                        }).await;

                        // Re-send start command to self
                        let self_tx = cmd_tx.clone();
                        let p = path.clone();
                        tokio::spawn(async move {
                            let _ = self_tx.send(WatcherCommand::Start { path: p }).await;
                        });
                    }
                }
            }
        }
    });

    (cmd_tx, event_rx)
}
```

```rust
/// Play a sound file asynchronously (fire and forget on a new thread).
pub fn play_sound(path: &Path) {
    let path = path.to_path_buf();
    thread::spawn(move || {
        if let Err(e) = play_sound_blocking(&path) {
            log::warn!("Failed to play sound {}: {}", path.display(), e);
        }
    });
}

fn play_sound_blocking(path: &Path) -> Result<(), Box<dyn std::error::Error>> {
    let (_stream, stream_handle) = OutputStream::try_default()?;
    let file = File::open(path)?;
    let source = Decoder::new(BufReader::new(file))?;
    let sink = Sink::try_new(&stream_handle)?;
    sink.append(source);
    sink.sleep_until_end();
    Ok(())
}

/// Resolve a sound path: if custom path is set and exists, use it. Otherwise use the bundled default.
```

```rust
}

/// Parse a hotkey string like "ctrl+F13" into modifiers and a virtual key code.
pub fn parse_hotkey(binding: &str) -> Result<HotkeyBinding, String> {
    let parts: Vec<&str> = binding.split('+').map(|s| s.trim()).collect();
    let mut ctrl = false;
    let mut alt = false;
    let mut shift = false;
    let mut vk_code: Option<u32> = None;

    for part in &parts {
        match part.to_lowercase().as_str() {
            "ctrl" | "control" => ctrl = true,
            "alt" => alt = true,
            "shift" => shift = true,
            key => {
                vk_code = Some(key_name_to_vk(key)?);
            }
        }
    }

    let vk = vk_code.ok_or_else(|| format!("No key specified in binding: {}", binding))?;
    Ok(HotkeyBinding { ctrl, alt, shift, vk_code: vk })
}

/// Convert a key name to a Windows virtual key code.
fn key_name_to_vk(name: &str) -> Result<u32, String> {
    match name.to_lowercase().as_str() {
        "f1" => Ok(0x70), "f2" => Ok(0x71), "f3" => Ok(0x72), "f4" => Ok(0x73),
        "f5" => Ok(0x74), "f6" => Ok(0x75), "f7" => Ok(0x76), "f8" => Ok(0x77),
        "f9" => Ok(0x78), "f10" => Ok(0x79), "f11" => Ok(0x7A), "f12" => Ok(0x7B),
        "f13" => Ok(0x7C), "f14" => Ok(0x7D), "f15" => Ok(0x7E), "f16" => Ok(0x7F),
        "f17" => Ok(0x80), "f18" => Ok(0x81), "f19" => Ok(0x82), "f20" => Ok(0x83),
        "f21" => Ok(0x84), "f22" => Ok(0x85), "f23" => Ok(0x86), "f24" => Ok(0x87),

        "5" => Ok(0x35), "6" => Ok(0x36), "7" => Ok(0x37), "8" => Ok(0x38),

        _ => Err(format!("Unknown key: {}", name)),
    }

/// Spawn the hotkey listener thread.
/// Uses Windows RegisterHotKey API (simpler than Raw Input for this use case).
/// Returns a receiver for hotkey events.
pub fn spawn_hotkey_listener(
    bindings: Vec<(HotkeyAction, String)>,
) -> Result<mpsc::Receiver<HotkeyAction>, String> {
    let (tx, rx) = mpsc::channel::<HotkeyAction>(32);

    let parsed: Vec<(HotkeyAction, HotkeyBinding)> = bindings
        .into_iter()
        .filter_map(|(action, bind_str)| {
            match parse_hotkey(&bind_str) {
                Ok(binding) => Some((action, binding)),
                Err(e) => {
                    log::warn!("Failed to parse hotkey '{}': {}", bind_str, e);
                    None
                }
            }
        })
        .collect();

    std::thread::spawn(move || {
        unsafe {
            use windows_sys::Win32::UI::Input::KeyboardAndMouse::*;
            use windows_sys::Win32::UI::WindowsAndMessaging::*;

            // Register each hotkey
            for (i, (_, binding)) in parsed.iter().enumerate() {
                let mut modifiers: u32 = MOD_NOREPEAT;
                if binding.ctrl { modifiers |= MOD_CONTROL; }
                if binding.alt { modifiers |= MOD_ALT; }
                if binding.shift { modifiers |= MOD_SHIFT; }

                let result = RegisterHotKey(
                    std::ptr::null_mut(),
                    (i + 1) as i32,
                    modifiers,
                    binding.vk_code,
                );

                if result == 0 {
                    log::warn!(
                        "Failed to register hotkey id={} vk=0x{:X}",
                        i + 1, binding.vk_code
                    );
                } else {
                    log::info!(
                        "Registered hotkey id={} vk=0x{:X} ctrl={} alt={} shift={}",
                        i + 1, binding.vk_code, binding.ctrl, binding.alt, binding.shift
                    );
                }
            }

            // Message pump
            let mut msg: MSG = std::mem::zeroed();
            while GetMessageW(&mut msg, std::ptr::null_mut(), 0, 0) > 0 {
                if msg.message == WM_HOTKEY {
                    let hotkey_id = msg.wParam as usize;
                    if hotkey_id > 0 && hotkey_id <= parsed.len() {
                        let action = parsed[hotkey_id - 1].0.clone();
                        let _ = tx.blocking_send(action);
                    }
                }
            }

            // Unregister on exit
            for i in 0..parsed.len() {
                UnregisterHotKey(std::ptr::null_mut(), (i + 1) as i32);
            }
        }
    });

    Ok(rx)
}
```

```rust
use base64::Engine;
use base64::engine::general_purpose::STANDARD as BASE64;
use futures_util::{SinkExt, StreamExt};
use sha2::{Digest, Sha256};
use tokio::sync::mpsc;
use tokio::time::{sleep, Duration};
use tokio_tungstenite::{connect_async, tungstenite::Message};

pub enum ObsWsEvent {
    Connected,
    Disconnected { code: Option<u16>, reason: String },
    ReplayBufferSaved { path: String },
    AuthError { message: String },
    Error { message: String },
    StatusChanged { status: String, attempt: Option<u32> },
}

pub enum ObsWsCommand {
    Connect,
    Disconnect,
}

fn build_auth_string(password: &str, salt: &str, challenge: &str) -> String {
    let secret_hash = Sha256::digest(format!("{}{}", password, salt).as_bytes());
    let secret = BASE64.encode(secret_hash);
    let auth_hash = Sha256::digest(format!("{}{}", secret, challenge).as_bytes());
    BASE64.encode(auth_hash)
}

/// Spawn the OBS WebSocket client task.
pub fn spawn_obs_ws(
    password: String,
    max_retries: u32,
) -> (mpsc::Sender<ObsWsCommand>, mpsc::Receiver<ObsWsEvent>) {
    let (cmd_tx, mut cmd_rx) = mpsc::channel::<ObsWsCommand>(8);
    let (event_tx, event_rx) = mpsc::channel::<ObsWsEvent>(64);

    tokio::spawn(async move {
        let mut retry_count: u32 = 0;

        while let Some(cmd) = cmd_rx.recv().await {
            match cmd {
                ObsWsCommand::Connect => {
                    retry_count = 0;
                    loop {
                        let _ = event_tx.send(ObsWsEvent::StatusChanged {
                            status: if retry_count == 0 { "connecting".into() } else { "reconnecting".into() },
                            attempt: if retry_count > 0 { Some(retry_count) } else { None },
                        }).await;

                        match connect_and_run(&password, &event_tx).await {
                            Ok(()) => {
                                retry_count = 0;
                                // Clean disconnect, don't retry
                                break;
                            }
                            Err(e) => {
                                retry_count += 1;
                                let _ = event_tx.send(ObsWsEvent::Error {
                                    message: format!("Connection failed: {}", e),
                                }).await;

                                if retry_count >= max_retries {
                                    let _ = event_tx.send(ObsWsEvent::StatusChanged {
                                        status: "failed-over".into(),
                                        attempt: Some(retry_count),
                                    }).await;
                                    break;
                                }

                                sleep(Duration::from_secs(5)).await;
                            }
                        }
                    }
                }
                ObsWsCommand::Disconnect => {
                    // Will be handled by dropping the connection
                    break;
                }
            }
        }
    });

    (cmd_tx, event_rx)
}

async fn connect_and_run(
    password: &str,
    event_tx: &mpsc::Sender<ObsWsEvent>,
) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
    let url = "ws://localhost:4455";
    let (ws_stream, _) = connect_async(url).await?;
    let (mut write, mut read) = ws_stream.split();

    let _ = event_tx.send(ObsWsEvent::Connected).await;

    while let Some(msg) = read.next().await {
        let msg = msg?;
        if let Message::Text(text) = msg {
            let data: serde_json::Value = serde_json::from_str(&text)?;
            let op = data.get("op").and_then(|v| v.as_u64()).unwrap_or(0);

            match op {
                // Hello (authentication challenge)
                0 => {
                    let d = &data["d"];
                    if let Some(auth) = d.get("authentication") {
                        let salt = auth["salt"].as_str().unwrap_or("");
                        let challenge = auth["challenge"].as_str().unwrap_or("");
                        let auth_string = build_auth_string(password, salt, challenge);

                        let payload = serde_json::json!({
                            "op": 1,
                            "d": {
                                "rpcVersion": 1,
                                "authentication": auth_string,
                                "eventSubscriptions": 1000
                            }
                        });
                        write.send(Message::Text(payload.to_string().into())).await?;
                    } else {
                        // No auth required
                        let payload = serde_json::json!({
                            "op": 1,
                            "d": {
                                "rpcVersion": 1,
                                "eventSubscriptions": 1000
                            }
                        });
                        write.send(Message::Text(payload.to_string().into())).await?;
                    }
                }
                // Identified (auth success)
                2 => {
                    let _ = event_tx.send(ObsWsEvent::StatusChanged {
                        status: "connected".into(),
                        attempt: None,
                    }).await;
                }
                // Event
                5 => {
                    let event_type = data["d"]["eventType"].as_str().unwrap_or("");
                    if event_type == "ReplayBufferSaved" {
                        if let Some(path) = data["d"]["eventData"]["savedReplayPath"].as_str() {
                            // Convert forward slashes to backslashes
                            let path = path.replace('/', "\\");
                            let _ = event_tx.send(ObsWsEvent::ReplayBufferSaved { path }).await;
                        }
                    }
                }
                _ => {}
            }
        }
    }

    let _ = event_tx.send(ObsWsEvent::Disconnected {
        code: None,
        reason: "Connection closed".into(),
    }).await;

    Ok(())
}
```

```rust
#[tauri::command]
pub fn press_gkey(
    key: u8,
    state: State<'_, AppState>,
    channels: State<'_, ChannelState>,
    app: AppHandle,
) -> Result<(), String> {
    let mut s = state.lock().map_err(|e| e.to_string())?;

    let gkey_label = format!("G{}", key);
    s.bind_chosen = Some(gkey_label.clone());

    // Log the keypress
    let entry = s.logger.log(
        LogLevel::Info,
        format!("{} Key", gkey_label),
        LogCategory::HotkeyPressed,
    );
    let _ = app.emit("log-entry", &entry);
    let _ = app.emit("hotkey-pressed", HotkeyPressedPayload { key: gkey_label.clone() });

    // Check if we have a current file
    let current = match &s.current_file {
        Some(cf) => cf.clone(),
        None => {
            let entry = s.logger.log(
                LogLevel::Error,
                "No current_file".into(),
                LogCategory::System,
            );
            let _ = app.emit("log-entry", &entry);

            // Play error sound
            let resource_dir = app.path().resource_dir().unwrap_or_default();
            if s.config.error_sound_enabled {
                sound::play_error(&s.config.error_sound_custom, &resource_dir);
            }
            return Ok(());
        }
    };

    // Get the file path to operate on (moved_path if available, else original)
    let file_path = current.moved_path.as_ref().unwrap_or(&current.path).clone();
    let config = s.config.clone();
    let log_enabled = config.log_file_enabled;
    let bind = s.bind_chosen.clone();

    // Release lock before file operations
    drop(s);

    // Perform the move/rename
    match mover::move_or_rename_file(&file_path, key, &config) {
        Ok(result) => {
            let mut s = state.lock().map_err(|e| e.to_string())?;

            // Update current file state
            if let Some(ref mut cf) = s.current_file {
                cf.moved_path = None; // File has been moved
            }
            s.current_file = Some(CurrentFile {
                path: result.new_path.clone(),
                moved_path: None,
                renamed: false,
            });

            let mode = if config.disable_file_movesorting { "renamed" } else { "moved" };
            let msg = format!("File {} to {}", mode, result.tag_applied);
            let entry = s.logger.log(LogLevel::Success, msg, LogCategory::FileMoved);
            let _ = app.emit("log-entry", &entry);
            let _ = app.emit("file-moved", FileMovedPayload {
                original: file_path.to_string_lossy().to_string(),
                destination: result.new_path.to_string_lossy().to_string(),
                tag: result.tag_applied.clone(),
                mode: mode.to_string(),
            });

            // Log to file
            if log_enabled {
                if let Some(ref bind) = bind {
                    let basename = file_path.file_name()
                        .and_then(|n| n.to_str())
                        .unwrap_or("unknown");
                    s.logger.write_to_file(&format!("{} | {}", bind, basename));
                }
            }

            // Play move sound
            let resource_dir = app.path().resource_dir().unwrap_or_default();
            if config.move_sound_enabled {
                sound::play_move_beep(&resource_dir);
            }
        }
        Err(e) => {
            let mut s = state.lock().map_err(|e| e.to_string())?;
            let entry = s.logger.log(LogLevel::Error, format!("Move failed: {}", e), LogCategory::System);
            let _ = app.emit("log-entry", &entry);
        }
    }

    Ok(())
}

#[tauri::command]
pub fn rename_file(
    text: String,
    state: State<'_, AppState>,
    app: AppHandle,
) -> Result<(), String> {
    let mut s = state.lock().map_err(|e| e.to_string())?;

    let current = match &s.current_file {
        Some(cf) => cf.clone(),
        None => {
            let entry = s.logger.log(LogLevel::Error, "No current_file".into(), LogCategory::System);
            let _ = app.emit("log-entry", &entry);
            return Ok(());
        }
    };

    let file_path = current.moved_path.as_ref().unwrap_or(&current.path).clone();
    let log_enabled = s.config.log_file_enabled;

    drop(s);

    match mover::rename_file_with_text(&file_path, &text) {
        Ok(result) => {
            let mut s = state.lock().map_err(|e| e.to_string())?;
            s.current_file = Some(CurrentFile {
                path: result.new_path.clone(),
                moved_path: Some(result.new_path.clone()),
                renamed: true,
            });

            let new_name = result.new_path.file_name()
                .and_then(|n| n.to_str())
                .unwrap_or("unknown")
                .to_string();

            let entry = s.logger.log(
                LogLevel::Success,
                format!("File renamed to: {}", new_name),
                LogCategory::FileRenamed,
            );
            let _ = app.emit("log-entry", &entry);
            let _ = app.emit("file-renamed", FileRenamedPayload {
                original: file_path.to_string_lossy().to_string(),
                new_name: new_name.clone(),
            });

            if log_enabled {
                let old_name = file_path.file_name()
                    .and_then(|n| n.to_str())
                    .unwrap_or("unknown");
                s.logger.write_to_file(&format!("Renamed: {} ---------> {}", old_name, new_name));
            }
        }
        Err(e) => {
            let mut s = state.lock().map_err(|e| e.to_string())?;
            let entry = s.logger.log(LogLevel::Error, format!("Rename failed: {}", e), LogCategory::System);
            let _ = app.emit("log-entry", &entry);
        }
    }

    Ok(())
}
```

```rust
/// Handle a new file creation event (from watcher or OBS WebSocket)
async fn handle_file_created(
    app: &tauri::AppHandle,
    state: &AppState,
    timer_tx: &tokio::sync::mpsc::Sender<TimerCommand>,
    config: &AppConfig,
    path: std::path::PathBuf,
) {
    // Check file size
    let size_mb = mover::file_size_mb(&path);
    let is_warning = !mover::is_file_size_valid(&path, 6.5);

    // Parse timestamp from filename
    let filename = path.file_name()
        .and_then(|n| n.to_str())
        .unwrap_or("unknown")
        .to_string();

    let timestamp = parse_time_from_filename(&filename);

    // Update state
    {
        let mut s = state.lock().unwrap();
        s.current_file = Some(CurrentFile {
            path: path.clone(),
            moved_path: Some(path.clone()),
            renamed: false,
        });
        s.bind_chosen = None;

        let msg = format!("{} | New file created: {}", timestamp, filename);
        let entry = s.logger.log(LogLevel::Info, msg, LogCategory::FileCreated);
        let _ = app.emit("log-entry", &entry);

        if is_warning {
            let warn_entry = s.logger.log(
                LogLevel::Warning,
                format!("{} | WARNING: File is {:.1}MB - probable black screen", timestamp, size_mb),
                LogCategory::FileCreated,
            );
            let _ = app.emit("log-entry", &warn_entry);
        }
    }

    // Emit file-created event
    let _ = app.emit("file-created", FileCreatedPayload {
        path: path.to_string_lossy().to_string(),
        filename,
        timestamp,
        size_mb,
        is_warning,
    });

    // Play sounds
    let resource_dir = app.path().resource_dir().unwrap_or_default();
    if config.clip_save_sound_enabled {
        sound::play_clip_saved(&config.clip_save_sound_custom, &resource_dir);
    }
    if is_warning && config.error_sound_enabled {
        sound::play_error(&config.error_sound_custom, &resource_dir);
    }

    // Start timer
    if config.timer_enabled {
        let duration_secs = config.timer_duration_secs() as u32;
        let _ = timer_tx.send(TimerCommand::Start { duration_secs }).await;
    }
}

fn parse_time_from_filename(filename: &str) -> String {
    use regex::Regex;

    // OBS format: YYYY-MM-DD HH-MM-SS
    let obs_re = Regex::new(r"(\d{4})-(\d{2})-(\d{2}) (\d{2})-(\d{2})-(\d{2})").unwrap();
    if let Some(caps) = obs_re.captures(filename) {
        let hour: u32 = caps[4].parse().unwrap_or(0);
        let min: u32 = caps[5].parse().unwrap_or(0);
        let (h12, ampm) = if hour == 0 { (12, "AM") }
            else if hour < 12 { (hour, "AM") }
            else if hour == 12 { (12, "PM") }
            else { (hour - 12, "PM") };
        return format!("{:02}:{:02} {}", h12, min, ampm);
    }

    "".to_string()
}
```

```typescript
import { useEffect, useState } from "react";
import { listen } from "@tauri-apps/api/event";
import { EVENTS } from "../lib/events";
import type { LogEntry } from "../types";

export function useEventLog() {
  const [entries, setEntries] = useState<LogEntry[]>([]);

  useEffect(() => {
    const unlisten = listen<LogEntry>(EVENTS.LOG_ENTRY, (event) => {
      setEntries((prev) => [...prev, event.payload]);
    });

    return () => {
      unlisten.then((fn) => fn());
    };
  }, []);

  const clear = () => setEntries([]);
  const restore = (restored: LogEntry[]) => setEntries(restored);

  return { entries, clear, restore };
}
```

```typescript
import { useEffect, useState } from "react";
import { listen } from "@tauri-apps/api/event";
import { EVENTS } from "../lib/events";
import type { TimerTick } from "../types";

interface TimerState {
  remainingSecs: number;
  totalSecs: number;
  running: boolean;
}

export function useTimer(initialDisplay: string) {
  const initialTotal = parseDisplay(initialDisplay);
  const [state, setState] = useState<TimerState>({
    remainingSecs: initialTotal,
    totalSecs: initialTotal,
    running: false,
  });

  useEffect(() => {
    const unlistenTick = listen<TimerTick>(EVENTS.TIMER_TICK, (event) => {
      setState({
        remainingSecs: event.payload.remainingSecs,
        totalSecs: event.payload.totalSecs,
        running: event.payload.remainingSecs > 0,
      });
    });

    const unlistenExpired = listen(EVENTS.TIMER_EXPIRED, () => {
      setState((prev) => ({ ...prev, remainingSecs: prev.totalSecs, running: false }));
    });

    return () => {
      unlistenTick.then((fn) => fn());
      unlistenExpired.then((fn) => fn());
    };
  }, []);

  return state;
}

function parseDisplay(display: string): number {
  const parts = display.split(":");
  if (parts.length === 2) {
    return parseInt(parts[0]) * 60 + parseInt(parts[1]);
  }
  return 70;
}
```

- [ ] **Step 4: Wire everything into final App.tsx**

Replace `src/App.tsx` with the full version integrating all components:

```tsx
import { useEffect, useState } from "react";
import { listen } from "@tauri-apps/api/event";
import { getConfig, updateConfig } from "./lib/commands";
import { EVENTS } from "./lib/events";
import { useEventLog } from "./hooks/useEventLog";
import { useTimer } from "./hooks/useTimer";
import { EventLog } from "./components/EventLog";
import { Sidebar } from "./components/Sidebar";
import { TimerDisplay } from "./components/TimerDisplay";
import { BottomBar } from "./components/BottomBar";
import { SettingsSheet } from "./components/SettingsSheet";
import { RenameDialog } from "./components/RenameDialog";
import type { AppConfig } from "./types";

function App() {
  const [config, setConfig] = useState<AppConfig | null>(null);
  const [settingsOpen, setSettingsOpen] = useState(false);
  const { entries, clear, restore } = useEventLog();

  useEffect(() => {
    getConfig().then(setConfig).catch(console.error);
  }, []);

  // Listen for config changes from backend
  useEffect(() => {
    const unlisten = listen<AppConfig>(EVENTS.CONFIG_CHANGED, (event) => {
      setConfig(event.payload);
    });
    return () => { unlisten.then((fn) => fn()); };
  }, []);

  // Listen for timer expiry to auto-wipe
  useEffect(() => {
    const unlisten = listen(EVENTS.TIMER_EXPIRED, () => {
      if (config?.auto_wipe_enabled) {
        clear();
      }
    });
    return () => { unlisten.then((fn) => fn()); };
  }, [config?.auto_wipe_enabled]);

  const timerDisplay = config
    ? `${String(Math.floor(config.timer_duration_ms / 60000)).padStart(2, "0")}:${String(Math.floor((config.timer_duration_ms % 60000) / 1000)).padStart(2, "0")}`
    : "01:10";
  const timer = useTimer(timerDisplay);

  if (!config) {
    return (
      <div className="flex h-screen items-center justify-center">
        <p className="text-muted-foreground">Loading...</p>
      </div>
    );
  }

  return (
    <div className="flex h-screen">
      <Sidebar onSettingsClick={() => setSettingsOpen(true)} />

      <main className="flex-1 flex flex-col min-w-0">
        <EventLog entries={entries} />
        <BottomBar
          mode={config.disable_file_movesorting ? "rename" : "sort"}
          autoWipe={config.auto_wipe_enabled}
          onAutoWipeChange={(v) => updateConfig({ auto_wipe_enabled: v }).then(setConfig)}
          onWipe={clear}
          onRestore={restore}
        />
      </main>

      <TimerDisplay
        remainingSecs={timer.remainingSecs}
        totalSecs={timer.totalSecs}
        running={timer.running}
      />

      <SettingsSheet
        open={settingsOpen}
        onOpenChange={setSettingsOpen}
        config={config}
        onConfigChange={setConfig}
      />

      <RenameDialog />
    </div>
  );
}

export default App;
```
